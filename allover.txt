2. Tech Stack Recommendation

2.1 RECOMMENDED: Hybrid Architecture

Frontend: Next.js 14+ (App Router)

Backend: Node.js + Express.js (separate server)

Real-time: Socket.io (on Express server)

Database: MongoDB with Mongoose

Why This Hybrid Approach?

1.  Next.js Benefits: Superior SEO for travel content, better
    performance with SSR/SSG, modern React features, easier deployment,
    built-in optimization

2.  Separate Backend Benefits: Full Socket.io support without custom
    server, better for complex real-time features, easier to scale
    independently, clearer separation of concerns

3.  Avoids WebSocket Issues: Using Next.js with custom server for
    WebSockets negates many Next.js benefits and complicates deployment.
    Separate Express server is cleaner.

// backend/src/middleware/auth.js
import jwt from 'jsonwebtoken';
export const authenticateToken = (req, res, next) => {
const token = req.header('Authorization')?.replace('Bearer ', '');
if (!token) {
return res.status(401).json({ error: 'Access denied' });
}
try {
const verified = jwt.verify(token, process.env.JWT_SECRET);
req.user = verified;
next();
} catch (error) {
res.status(400).json({ error: 'Invalid token' });
}
};

import OpenAI from 'openai';
const openai = new OpenAI({
apiKey: process.env.OPENAI_API_KEY

export async function generateItinerary(preferences) {
const prompt = `Generate a detailed travel itinerary based on:
- Destination: ${preferences.destination}
- Duration: ${preferences.days} days
- Budget: ${preferences.budget}
- Interests: ${preferences.interests.join(', ')}
- Dietary: ${preferences.dietary || 'none'}
Include:
1. Day-by-day schedule
2. Restaurant recommendations (halal/vegan if specified)
3. UNESCO sites if applicable
4. Best times to visit attractions
5. Estimated costs
Return as JSON.`;
const response = await openai.chat.completions.create({
model: 'gpt-4o-mini',
messages: [{ role: 'user', content: prompt }],
response_format: { type: 'json_object' }
});
return JSON.parse(response.choices[0].message.content);
}

// frontend/store/authStore.ts
import { create } from 'zustand';
import { persist } from 'zustand/middleware';
interface AuthState {
user: User | null;
token: string | null;
setUser: (user: User, token: string) => void;
logout: () => void;
}
export const useAuthStore = create<AuthState>()(persist(
(set) => ({
user: null,
token: null,
setUser: (user, token) => set({ user, token }),
logout: () => set({ user: null, token: null })
}),
{ name: 'auth-storage' }
));

Step 5.3: API Client Setup

// frontend/lib/api.ts
import axios from 'axios';
import { useAuthStore } from '@/store/authStore';
const api = axios.create({
baseURL: process.env.NEXT_PUBLIC_API_URL
});
// Add auth token to requests
api.interceptors.request.use((config) => {
const token = useAuthStore.getState().token;
if (token) {
config.headers.Authorization = `Bearer ${token}`;
}
return config;
});
// Handle auth errors
api.interceptors.response.use(
(response) => response,
(error) => {
if (error.response?.status === 401) {
useAuthStore.getState().logout();
window.location.href = '/auth/login';
}
return Promise.reject(error);
}
);
export default api;

Phase 6: Real-time Features (Week 17-18)

Step 6.1: Socket.io Backend Setup

// backend/src/sockets/index.js
import { Server } from 'socket.io';
export function setupSockets(httpServer) {
const io = new Server(httpServer, {
cors: {
origin: process.env.FRONTEND_URL,
credentials: true
}
});
io.on('connection', (socket) => {
console.log('User connected:', socket.id);
// Location tracking
socket.on('location:update', async (data) => {
const { userId, location } = data;
socket.to(`group:${data.groupId}`).emit('location:updated', {
userId,
location,
timestamp: Date.now()
});
});
// Group chat
socket.on('message:send', async (data) => {
io.to(`group:${data.groupId}`).emit('message:new', data);
});
// Join group room
socket.on('group:join', (groupId) => {
socket.join(`group:${groupId}`);
});
socket.on('disconnect', () => {
console.log('User disconnected:', socket.id);
});
});
return io;
}

Step 6.2: Frontend Socket Integration

6. Best Practices & Optimization

6.1 Performance Optimization

-   Server-Side Rendering: Use Next.js SSR for SEO-critical pages
    (destinations, itineraries)

-   Image Optimization: Use Next.js Image component, Cloudinary
    auto-optimization

-   Code Splitting: Dynamic imports for heavy components (maps, charts)

-   API Response Caching: Redis for OpenAI responses, weather data,
    currency rates

-   Database Indexing: Index frequently queried fields (userId,
    destination, dates)

6.2 Security Best Practices

-   Environment Variables: Never commit .env files, use .env.example
    template

-   Rate Limiting: Implement express-rate-limit on API endpoints

-   Input Validation: Use Zod schemas on both frontend and backend

-   CORS Configuration: Whitelist only your frontend domain in
    production

-   HTTPS Only: Enforce HTTPS in production, use secure cookies

6.3 Cost Optimization

-   OpenAI Costs: Use GPT-4o-mini instead of GPT-4, implement aggressive
    caching, set max_tokens limits

-   API Rate Limits: Respect free tier limits, implement request queuing

-   Database: Use MongoDB Atlas free tier initially, monitor usage

-   Hosting: Vercel for Next.js (free tier), Railway/Render for backend

8. Common Pitfalls to Avoid

17. WebSocket Issues in Next.js: Don't use Next.js custom server for
    Socket.io. Use separate Express backend as recommended.

18. API Key Exposure: Never expose API keys in frontend code. Always
    proxy through backend.

19. CORS Errors: Configure CORS properly on backend, ensure credentials
    are handled correctly.

20. OpenAI Cost Overruns: Implement caching, rate limiting, and use
    GPT-4o-mini for cost efficiency.

21. State Management Complexity: Keep it simple with Zustand. Avoid
    Redux unless absolutely necessary.

22. Real-time Connection Management: Handle Socket.io reconnection
    logic, implement heartbeat mechanism.

23. Mobile Responsiveness: Design mobile-first, test on actual devices,
    not just browser DevTools.

9. Testing Strategy

9.1 Backend Testing

-   Unit Tests: Jest for services, utilities, helpers

-   Integration Tests: Supertest for API endpoints

-   Database Tests: Use MongoDB Memory Server for isolated tests

9.2 Frontend Testing

-   Component Tests: React Testing Library

-   E2E Tests: Playwright for critical user flows

-   Visual Regression: Chromatic for UI consistency

Understanding the Knowledge Graph

Why Knowledge Graph?

-   Understands RELATIONSHIPS between entities (restaurants NEAR
    attractions)

-   Better than keyword search for complex queries

-   Can traverse connections (find halal food near family-friendly
    parks)

-   Learns from user interactions

Graph Schema Design:

NODES (Entities):
- Destination (cities, countries)
- POI (points of interest: restaurants, hotels, attractions)
- Activity (things to do)
- Tag (halal, vegan, family-friendly, budget, luxury)
- User (stored here for graph queries)
RELATIONSHIPS:
- (POI)-[:LOCATED_IN]->(Destination)
- (POI)-[:NEAR {distance: 500}]->(POI)
- (POI)-[:OFFERS]->(Activity)
- (POI)-[:HAS_TAG]->(Tag)
- (User)-[:VISITED]->(POI)
- (User)-[:PREFERS]->(Tag)
- (Destination)-[:SIMILAR_TO {score: 0.8}]->(Destination)

/**
* Add destination to graph
*/
export async function addDestination(data) {
const { name, country, lat, lng, description } = data;
const result = await runQuery(`
MERGE (d:Destination {name: $name})
SET d.country = $country,
d.lat = $lat,
d.lng = $lng,
d.description = $description,
d.updatedAt = datetime()
RETURN d
`, { name, country, lat, lng, description });
return result[0]?.get('d').properties;
}
/**
* Add POI (restaurant, attraction, hotel)
*/
export async function addPOI(data) {
const { name, type, destination, lat, lng, description, tags = [] } =
data;
await runQuery(`
// Create POI
MERGE (poi:POI {name: $name})
SET poi.type = $type,
poi.lat = $lat,
poi.lng = $lng,
poi.description = $description,
poi.updatedAt = datetime()
// Link to destination
MERGE (dest:Destination {name: $destination})
MERGE (poi)-[:LOCATED_IN]->(dest)
// Add tags
WITH poi
UNWIND $tags as tagName
MERGE (tag:Tag {name: tagName})
MERGE (poi)-[:HAS_TAG]->(tag)
`, { name, type, destination, lat, lng, description, tags });
}
/**
* Create proximity relationships
* Finds POIs within distance and creates NEAR relationships
*/
export async function createProximityRelationships(maxDistance = 2000) {
await runQuery(`
MATCH (p1:POI), (p2:POI)
WHERE id(p1) < id(p2)
WITH p1, p2,
point.distance(
point({latitude: p1.lat, longitude: p1.lng}),
point({latitude: p2.lat, longitude: p2.lng})
) AS dist
WHERE dist < $maxDistance
MERGE (p1)-[r:NEAR]->(p2)
SET r.distance = dist
`, { maxDistance });
console.log('✅ Proximity relationships created');
}
/**
* Search with Graph RAG
* Combines semantic search with graph traversal
*/
export async function graphRAGSearch(query, filters = {}) {
const { destination, tags = [], type, maxResults = 10 } = filters;
let cypher = `
MATCH (poi:POI)
`;
// Add destination filter
if (destination) {
cypher += `
-[:LOCATED_IN]->(dest:Destination {name: $destination})
`;
}
// Add tag filters
if (tags.length > 0) {
cypher += `
MATCH (poi)-[:HAS_TAG]->(tag:Tag)
WHERE tag.name IN $tags
`;
}
// Add type filter
if (type) {
cypher += `
WHERE poi.type = $type
`;
}
cypher += `
OPTIONAL MATCH (poi)-[:LOCATED_IN]->(dest:Destination)
OPTIONAL MATCH (poi)-[:HAS_TAG]->(tag:Tag)
OPTIONAL MATCH (poi)-[near:NEAR]-(nearby:POI)
WHERE near.distance < 1000
RETURN poi,
dest,
collect(DISTINCT tag.name) as tags,
collect(DISTINCT {
name: nearby.name,
type: nearby.type,
distance: near.distance
})[0..5] as nearbyPOIs
LIMIT $maxResults
`;
const results = await runQuery(cypher, {
destination,
tags,
type,
maxResults
});
return results.map(record => ({
poi: record.get('poi').properties,
destination: record.get('dest')?.properties,
tags: record.get('tags'),
nearby: record.get('nearbyPOIs')
}));
}
/**
* Find halal food near location
*/
export async function findHalalNearLocation(location, radius = 5000) {
const { lat, lng } = location;
const results = await runQuery(`
MATCH (poi:POI)-[:HAS_TAG]->(halal:Tag {name: "halal"})
WHERE poi.type = "restaurant"
WITH poi,
point.distance(
point({latitude: $lat, longitude: $lng}),
point({latitude: poi.lat, longitude: poi.lng})
) AS distance
WHERE distance < $radius
OPTIONAL MATCH (poi)-[:LOCATED_IN]->(dest:Destination)
OPTIONAL MATCH (poi)-[near:NEAR]-(nearby:POI)
WHERE near.distance < 500 AND nearby.type = "attraction"
RETURN poi,
dest,
distance,
collect(DISTINCT nearby.name)[0..3] as nearbyAttractions
ORDER BY distance ASC
LIMIT 20
`, { lat, lng, radius });
return results.map(record => ({
restaurant: record.get('poi').properties,
destination: record.get('dest')?.properties,
distance: Math.round(record.get('distance').low),
nearbyAttractions: record.get('nearbyAttractions')
}));
}
/**
* Get personalized recommendations based on user preferences
*/
export async function getPersonalizedRecommendations(userId,
userPreferences, destination) {
const { dietary = [], interests = [] } = userPreferences;
// Combine dietary and interests as tags to search for
const preferredTags = [...dietary, ...interests];
const results = await runQuery(`
MATCH (poi:POI)-[:LOCATED_IN]->(dest:Destination {name: $destination})
OPTIONAL MATCH (poi)-[:HAS_TAG]->(tag:Tag)
WHERE tag.name IN $preferredTags
WITH poi, collect(DISTINCT tag.name) as matchedTags
WHERE size(matchedTags) > 0
OPTIONAL MATCH (poi)-[:LOCATED_IN]->(dest:Destination)
OPTIONAL MATCH (poi)-[near:NEAR]-(nearby:POI)
RETURN poi,
dest,
matchedTags,
size(matchedTags) as relevanceScore,
collect(DISTINCT nearby.name)[0..3] as nearbyPOIs
ORDER BY relevanceScore DESC
LIMIT 20
`, { destination, preferredTags });
return results.map(record => ({
poi: record.get('poi').properties,
destination: record.get('dest')?.properties,
matchedPreferences: record.get('matchedTags'),
relevanceScore: record.get('relevanceScore').low,
nearby: record.get('nearbyPOIs')
}));
}
